#include <iostream>
using namespace std;

int main() {
    int choice;
    setlocale(LC_ALL, "RU");
    // Выводим меню
    cout << "[ - ] Программа \"Времена года\"" << endl;
    cout << endl;
    cout << "[ 1 ] Лето" << endl;
    cout << "[ 2 ] Осень" << endl;
    cout << "[ 3 ] Зима" << endl;
    cout << "[ 4 ] Весна" << endl;
    cout << endl;
    cout << "[ - ] Введите значение: ";

    // Считываем выбор пользователя
    cin >> choice;

    // Обрабатываем выбор с помощью if-else if-else
    if (choice == 1) {
        setlocale(LC_ALL, "RU");
        cout << "[ - ] Сейчас лето, можно одеться легко и взять с собой воду." << endl;
    }
    else if (choice == 2) {
        cout << "[ - ] Сейчас осень, следует одеться тепло и взять с собой зонт." << endl;
    }
    else if (choice == 3) {
        cout << "[ - ] Сейчас зима, нужно тепло одеться и не забыть варежки." << endl;
    }
    else if (choice == 4) {
        cout << "[ - ] Сейчас весна, можно одеться полегче, но взять с собой куртку." << endl;
    }
    else {
        cout << "[ - ] Ошибка: такого времени года нет! Пожалуйста, выберите от 1 до 4." << endl;
    }

    return 0;
}
